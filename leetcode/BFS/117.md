
# 填充每个节点的下一个右侧节点指针 II

## 树的层序遍历

```c++
/*
// Definition for a Node.
class Node {
public:
    int val;
    Node* left;
    Node* right;
    Node* next;

    Node() : val(0), left(NULL), right(NULL), next(NULL) {}

    Node(int _val) : val(_val), left(NULL), right(NULL), next(NULL) {}

    Node(int _val, Node* _left, Node* _right, Node* _next)
        : val(_val), left(_left), right(_right), next(_next) {}
};
*/

class Solution {
public:
    Node* connect(Node* root) {

        if(root == NULL ) return root;
        
        queue<Node*> q;

        int n = 1;
        q.push(root);

        while(!q.empty()){

            int nChild = 0;

            Node *cur = NULL;

            for(int i = 0; i < n; i++){
                Node *n = q.front();
                q.pop();

                if(cur == NULL){
                    cur = n;
                }else{
                    cur = cur->next = n;
                }

                if(n->left != NULL){
                    nChild++;
                    q.push(n->left);
                }
                if(n->right != NULL){
                    nChild++;
                    q.push(n->right);
                }
            }

            n = nChild;
        }

        return root;
    }
};
```
