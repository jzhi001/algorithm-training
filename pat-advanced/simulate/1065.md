#include <iostream>
#include <climits>

using namespace std;

int n;

bool cmp(long long a, long long b, long long c){

    if(a > 0 && b > 0 && a > LONG_MAX - b){
        return true;
    }else if(a < 0 && b < 0 && a < LONG_MIN - b){
        return false;
    }else{
        return a + b > c;
    }
}

int main(){

    cin >> n;

    for(int i = 1; i <= n; i++){
        long long a, b, c;
        cin >> a >> b >> c;

        bool ans = cmp(a, b, c);

        if(i > 1) cout << endl;
        printf("Case #%d: %s", i, (ans ? "true" : "false"));
    }

    return 0;
}
